       def json_diff(old: dict | None, new: dict) -> list[dict]:
           """Compute RFC 6902-style JSON patch operations between old and new dicts."""
           - If old is None: return [{"op": "add", "path": "", "value": new}]
           - For each key in union of old/new keys (sorted for determinism):
             - Key in new only: {"op": "add", "path": "/{key}", "value": new[key]}
             - Key in old only: {"op": "remove", "path": "/{key}"}
             - Values differ and both dicts: recurse with path prefix
             - Values differ otherwise: {"op": "replace", "path": "/{key}", "value": new[key]}
           - Handle nested dicts recursively, arrays as atomic replace (not element-level)
           - Use the pattern from RESEARCH.md

       def apply_patch(base: dict, patch: list[dict]) -> dict:
           """Apply RFC 6902 patch operations to a base dict. For version reconstruction."""
           - Handle ops: add, remove, replace
           - Navigate nested paths (split on "/")
           - Return new dict (do not mutate base)

    2. Create scripts/change_journal.py -- DREG-06:

       class ChangeJournal:
           def __init__(self, journal_path: str):
               self._path = journal_path

           def append(self, slot_id: str, slot_type: str, operation: str,
                      version_before: int, version_after: int, agent_id: str,
                      summary: str, old_content: dict | None, new_content: dict | None) -> dict:
               """Append a journal entry. Compute RFC 6902 diff. fsync after write."""
               - Build entry dict: timestamp (UTC ISO 8601), slot_id, slot_type, operation,
                 version_before, version_after, agent_id, summary, diff (from json_diff)
               - Serialize as compact JSON (separators=(",",":")) + newline
               - Append to file, flush, fsync (per RESEARCH.md pattern)
               - Return the entry dict

           def query_by_slot(self, slot_id: str) -> list[dict]:
               """Return all journal entries for a given slot_id, ordered by timestamp."""

           def query_time_range(self, start: str, end: str) -> list[dict]:
               """Return entries within UTC time range [start, end] inclusive."""
               - Per Pitfall 2: wrap each line parse in try/except
               - If last line is corrupt, log warning and skip
               - Compare timestamps as strings (ISO 8601 is lexicographically sortable)

           def query_all(self) -> list[dict]:
               """Return all journal entries. Handle corrupt last line gracefully."""

    3. Create scripts/version_manager.py -- DREG-05:

       class VersionManager:
           def __init__(self, journal: ChangeJournal):
               self._journal = journal

           def get_history(self, slot_id: str) -> list[dict]:
               """Return version history for a slot: list of {version, timestamp, agent_id, summary, operation}."""
               - Query journal for slot_id
               - Extract version metadata from each entry
               - Return ordered list

           def get_version(self, slot_id: str, version: int, current_content: dict) -> dict | None:
               """Reconstruct a specific version by reverse-applying diffs from current."""
               - Get journal entries for slot_id, sorted by version descending
               - Starting from current_content, reverse-apply patches back to target version
               - If journal entries are missing (truncated), return None with warning
               - Per Pitfall 1: return "history unavailable" rather than failing

           def get_current_version(self, slot_id: str) -> int | None:
               """Return the latest version number from journal entries. None if no entries."""
