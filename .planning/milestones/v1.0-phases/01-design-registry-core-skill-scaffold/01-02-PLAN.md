    class SchemaValidator:
        def __init__(self, schemas_dir: str):
            """Load all JSON Schema files from schemas/ directory into a dict keyed by slot_type."""
            - Read each .json file in schemas_dir
            - Parse and store as {slot_type_name: schema_dict}
            - slot_type_name derived from filename (e.g., "component" from "component.json", "requirement-ref" from "requirement-ref.json")

        def validate(self, slot_type: str, content: dict) -> list[dict]:
            """Validate content against schema for slot_type. Returns empty list if valid, list of error dicts if invalid."""
            - Look up schema by slot_type; raise ValueError if unknown type
            - Use jsonschema.Draft202012Validator(schema)
            - Iterate errors with validator.iter_errors(content), sorted by path
            - For each error, produce dict: {"path": field_path, "constraint": error.validator, "message": error.message, "hint": _generate_hint(error), "actual_value": ...}
            - Per user decision: "detailed with exact field path + fix hint"

        def validate_or_raise(self, slot_type: str, content: dict) -> None:
            """Validate and raise SchemaValidationError if invalid."""
            errors = self.validate(slot_type, content)
            if errors:
                raise SchemaValidationError(slot_type, errors)

        @property
        def supported_types(self) -> list[str]:
            """Return list of supported slot types."""

    class SchemaValidationError(Exception):
        def __init__(self, slot_type: str, errors: list[dict]):
            self.slot_type = slot_type
            self.errors = errors
            msg = f"Schema validation failed for '{slot_type}': {len(errors)} error(s)\n"
            for e in errors:
                msg += f"  - {e['path']}: {e['message']}. Hint: {e['hint']}\n"
            super().__init__(msg)

    Helper function _generate_hint(error) per RESEARCH.md pattern:
    - required -> "Add the '{field}' field to this object."
    - type -> "Change the value to type '{expected}'."
    - enum -> "Use one of: {values}."
    - pattern -> "Value must match pattern: {pattern}"
    - minLength -> "Value must be at least {min} characters."
    - const -> "Value must be exactly '{value}'."
    - additionalProperties -> "Remove unexpected field '{field}'. Allowed fields: {properties}."
    - default -> "Check the schema definition for valid values."

    class SlotStorageEngine:
        def __init__(self, workspace_root: str):
            """Initialize with .system-dev/ path. Validate workspace exists."""
            self._root = workspace_root  # path to .system-dev/
            self._registry_dir = os.path.join(workspace_root, "registry")
            self._index_path = os.path.join(workspace_root, "index.json")

        def read(self, slot_id: str) -> dict | None:
            """Read a slot by ID. Look up path in index. Return None if not found."""
            - Load index.json, find slot_id in slots dict
            - Read the slot JSON file from the path in index
            - Return parsed dict or None

        def write(self, slot_id: str, slot_type: str, content: dict) -> dict:
            """Write slot file atomically. Update index. Return {"path": relative_path}."""
            - Determine file path: registry/{type_dir}/{slot_id}.json
            - validate_path() against workspace root (path traversal protection)
            - atomic_write() the content dict
            - Update index.json: add/update entry with path, slot_type, version, updated_at
            - atomic_write() the updated index
            - Return {"path": relative_path}

        def delete(self, slot_id: str) -> bool:
            """Delete slot file and remove from index. Return True if deleted, False if not found."""

        def query(self, slot_type: str, filters: dict | None = None) -> list[dict]:
            """Query slots by type. Optional field=value filters applied to slot content."""
            - Scan index for matching slot_type
            - Load each matching slot file
            - Apply filters (simple equality on top-level fields)
            - Return list of slot dicts

        def rebuild_index(self) -> dict:
            """Rebuild index.json from filesystem scan. Source of truth is files on disk per Pitfall 3."""
            - Walk registry/ subdirectories
            - Read each .json file, extract slot_id, slot_type, version, updated_at
            - Write new index.json atomically
            - Return {"rebuilt": count, "removed_stale": count}

    class SlotAPI:
        def __init__(self, workspace_root: str, schemas_dir: str):
            """Initialize all subsystems. This is the ONLY entry point for agents (XCUT-04)."""
            self._storage = SlotStorageEngine(workspace_root)
            self._validator = SchemaValidator(schemas_dir)
            # version_manager and change_journal will be added in Plan 03
            # For now, SlotAPI handles versioning inline (set version field)

        def create(self, slot_type: str, content: dict, agent_id: str = "user") -> dict:
            """Create a new slot. Generate ID, set version=1, validate, persist."""
            - Generate slot_id: f"{prefix}-{uuid4()}"
            - Set content["slot_id"] = slot_id
            - Set content["slot_type"] = slot_type
            - Set content["version"] = 1
            - Set content["created_at"] = content["updated_at"] = datetime.now(UTC).isoformat()
            - self._validator.validate_or_raise(slot_type, content)
            - self._storage.write(slot_id, slot_type, content)
            - Return {"status": "created", "slot_id": slot_id, "version": 1}

        def read(self, slot_id: str) -> dict | None:
            """Read slot by ID. Delegates to storage engine."""

        def update(self, slot_id: str, content: dict, expected_version: int | None = None, agent_id: str = "user") -> dict:
            """Update existing slot. Optimistic locking: reject if expected_version != current."""
            - Read current slot; raise if not found
            - If expected_version provided and != current["version"], raise ConflictError
            - Increment version: content["version"] = current["version"] + 1
            - Preserve slot_id, slot_type, created_at from current
            - Set updated_at to now
            - Validate against schema
            - Persist
            - Return {"status": "updated", "slot_id": slot_id, "version": new_version}

        def delete(self, slot_id: str, agent_id: str = "user") -> dict:
            """Delete slot by ID."""

        def query(self, slot_type: str, filters: dict | None = None) -> list[dict]:
            """Query slots by type with optional filters."""

    class ConflictError(Exception):
        """Raised when optimistic locking detects version mismatch."""
        def __init__(self, slot_id: str, expected: int, actual: int):
            super().__init__(f"Version conflict for '{slot_id}': expected v{expected}, found v{actual}")
            self.slot_id = slot_id
            self.expected_version = expected
            self.actual_version = actual
